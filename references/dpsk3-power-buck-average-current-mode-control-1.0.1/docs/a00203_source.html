128 void drv_LcdInterface_SpiSend(uint8_t data)
129 {
130  uint8_t mask;
131 
132  for(mask = 0x80; mask != 0; mask >>= 1)
133  {
134  LCD_SCL_SetLow();
135  Nop();
136  __delay_us(1);
137  if(data & mask)
138  LCD_SDI_SetHigh();
139  else
140  LCD_SDI_SetLow();
141  LCD_SCL_SetHigh();
142  Nop();
143  __delay_us(1);
144  }
145  LCD_CS_nSELECTED();
146  __delay_us(50);
147 }
